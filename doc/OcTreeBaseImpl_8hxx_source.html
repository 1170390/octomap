00415   template <class NODE,class I>
00416   bool OcTreeBaseImpl<NODE,I>::computeRayKeys(const point3d& origin,
00417                                           const point3d& end, 
00418                                           KeyRay& ray) const {
00419 
00420     // see "A Faster Voxel Traversal Algorithm for Ray Tracing" by Amanatides & Woo
00421     // basically: DDA in 3D
00422 
00423     ray.reset();
00424 
00425     OcTreeKey key_origin, key_end;
00426     if ( !OcTreeBaseImpl<NODE,I>::coordToKeyChecked(origin, key_origin) ||
00427          !OcTreeBaseImpl<NODE,I>::coordToKeyChecked(end, key_end) ) {
00428       OCTOMAP_WARNING_STR("coordinates ( "
00429                 << origin << " -> " << end << ") out of bounds in computeRayKeys");
00430       return false;
00431     }
00432 
00433     
00434     if (key_origin == key_end) return true; // same tree cell, we're done.
00435 
00436     ray.addKey(key_origin);
00437 
00438     // Initialization phase -------------------------------------------------------
00439 
00440     point3d direction = (end - origin);
00441     float length = (float) direction.norm();
00442     direction /= length; // normalize vector
00443 
00444     int    step[3];
00445     double tMax[3];
00446     double tDelta[3];
00447 
00448     OcTreeKey current_key = key_origin; 
00449 
00450     for(unsigned int i=0; i < 3; ++i) {
00451       // compute step direction
00452       if (direction(i) > 0.0) step[i] =  1;
00453       else if (direction(i) < 0.0)   step[i] = -1;
00454       else step[i] = 0;
00455 
00456       // compute tMax, tDelta
00457       if (step[i] != 0) {
00458         // corner point of voxel (in direction of ray)
00459         double voxelBorder = this->keyToCoord(current_key[i]);
00460         voxelBorder += (float) (step[i] * this->resolution * 0.5);
00461 
00462         tMax[i] = ( voxelBorder - origin(i) ) / direction(i);
00463         tDelta[i] = this->resolution / fabs( direction(i) );
00464       }
00465       else {
00466         tMax[i] =  std::numeric_limits<double>::max( );
00467         tDelta[i] = std::numeric_limits<double>::max( );
00468       }
00469     }
00470 
00471     // for speedup:
00472     point3d origin_scaled = origin;  
00473     origin_scaled /= (float) this->resolution;  
00474     double length_scaled = length - this->resolution/4.; // safety margin
00475     length_scaled /= this->resolution;  // scale 
00476     length_scaled = length_scaled*length_scaled;  // avoid sqrt in dist comp.
00477 
00478     // Incremental phase  ---------------------------------------------------------
00479 
00480     bool done = false;
00481     while (!done) {
00482 
00483       unsigned int dim;
00484 
00485       // find minimum tMax:
00486       if (tMax[0] < tMax[1]){
00487         if (tMax[0] < tMax[2]) dim = 0;
00488         else                   dim = 2;
00489       }
00490       else {
00491         if (tMax[1] < tMax[2]) dim = 1;
00492         else                   dim = 2;
00493       }
00494 
00495       // advance in direction "dim"
00496       current_key[dim] += step[dim];
00497       tMax[dim] += tDelta[dim];
00498 
00499       assert (current_key[dim] < 2*this->tree_max_val);
00500 
00501       // reached endpoint, key equv?
00502       if (current_key == key_end) {
00503         done = true;
00504         break;
00505       }
00506       else {
00507 
00508         // reached endpoint world coords?
00509         double dist_from_origin = 0;
00510         for (unsigned int j = 0; j < 3; j++) {
00511           double coord = (double) current_key[j] - (double) this->tree_max_val;
00512           dist_from_origin += (coord - origin_scaled(j)) * (coord - origin_scaled(j));
00513         }
00514         if (dist_from_origin > length_scaled) {
00515           done = true;
00516           break;
00517         }
00518         
00519         else {  // continue to add freespace cells
00520           ray.addKey(current_key);
00521         }
00522       }
00523 
00524       assert ( ray.size() < ray.sizeMax() - 1);
00525       
00526     } // end while
00527 
00528     return true;
00529   }
