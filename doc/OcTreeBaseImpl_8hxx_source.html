410  template <class NODE,class I>
411  bool OcTreeBaseImpl<NODE,I>::computeRayKeys(const point3d& origin,
412  const point3d& end,
413  KeyRay& ray) const {
414 
415  // see "A Faster Voxel Traversal Algorithm for Ray Tracing" by Amanatides & Woo
416  // basically: DDA in 3D
417 
418  ray.reset();
419 
420  OcTreeKey key_origin, key_end;
421  if ( !OcTreeBaseImpl<NODE,I>::coordToKeyChecked(origin, key_origin) ||
422  !OcTreeBaseImpl<NODE,I>::coordToKeyChecked(end, key_end) ) {
423  OCTOMAP_WARNING_STR("coordinates ( "
424  << origin << " -> " << end << ") out of bounds in computeRayKeys");
425  return false;
426  }
427 
428 
429  if (key_origin == key_end)
430  return true; // same tree cell, we're done.
431 
432  ray.addKey(key_origin);
433 
434  // Initialization phase -------------------------------------------------------
435 
436  point3d direction = (end - origin);
437  float length = (float) direction.norm();
438  direction /= length; // normalize vector
439 
440  int step[3];
441  double tMax[3];
442  double tDelta[3];
443 
444  OcTreeKey current_key = key_origin;
445 
446  for(unsigned int i=0; i < 3; ++i) {
447  // compute step direction
448  if (direction(i) > 0.0) step[i] = 1;
449  else if (direction(i) < 0.0) step[i] = -1;
450  else step[i] = 0;
451 
452  // compute tMax, tDelta
453  if (step[i] != 0) {
454  // corner point of voxel (in direction of ray)
455  double voxelBorder = this->keyToCoord(current_key[i]);
456  voxelBorder += (float) (step[i] * this->resolution * 0.5);
457 
458  tMax[i] = ( voxelBorder - origin(i) ) / direction(i);
459  tDelta[i] = this->resolution / fabs( direction(i) );
460  }
461  else {
462  tMax[i] = std::numeric_limits<double>::max( );
463  tDelta[i] = std::numeric_limits<double>::max( );
464  }
465  }
466 
467  // for speedup:
468  point3d origin_scaled = origin;
469  origin_scaled /= (float) this->resolution;
470  double length_scaled = length - this->resolution/4.; // safety margin
471  length_scaled /= this->resolution; // scale
472  length_scaled = length_scaled*length_scaled; // avoid sqrt in dist comp.
473 
474  // Incremental phase ---------------------------------------------------------
475 
476  bool done = false;
477  while (!done) {
478 
479  unsigned int dim;
480 
481  // find minimum tMax:
482  if (tMax[0] < tMax[1]){
483  if (tMax[0] < tMax[2]) dim = 0;
484  else dim = 2;
485  }
486  else {
487  if (tMax[1] < tMax[2]) dim = 1;
488  else dim = 2;
489  }
490 
491  // advance in direction "dim"
492  current_key[dim] += step[dim];
493  tMax[dim] += tDelta[dim];
494 
495  assert (current_key[dim] < 2*this->tree_max_val);
496 
497  // reached endpoint, key equv?
498  if (current_key == key_end) {
499  done = true;
500  break;
501  }
502  else {
503 
504  // reached endpoint world coords?
505  double dist_from_origin = 0;
506  for (unsigned int j = 0; j < 3; j++) {
507  double coord = (double) current_key[j] - (double) this->tree_max_val;
508  dist_from_origin += (coord - origin_scaled(j)) * (coord - origin_scaled(j));
509  }
510  if (dist_from_origin > length_scaled) {
511  done = true;
512  break;
513  }
514 
515  else { // continue to add freespace cells
516  ray.addKey(current_key);
517  }
518  }
519 
520  assert ( ray.size() < ray.sizeMax() - 1);
521 
522  } // end while
523 
524  return true;
525  }

539  bool OcTreeBaseImpl<NODE,I>::deleteNodeRecurs(NODE* node, unsigned int depth, unsigned int max_depth, const OcTreeKey& key){
540  if (depth >= max_depth) // on last level: delete child when going up
541  return true;
542 
543  assert(node);
544 
545  unsigned int pos = computeChildIdx(key, this->tree_depth-1-depth);
546 
547  if (!node->childExists(pos)) {
548  // child does not exist, but maybe it's a pruned node?
549  if ((!node->hasChildren()) && (node != this->root)) {
550  // current node does not have children AND it's not the root node
551  // -> expand pruned node
552  node->expandNode();
553  this->tree_size+=8;
554  this->size_changed = true;
555  } else { // no branch here, node does not exist
556  return false;
557  }
558  }
559 
560  // follow down further, fix inner nodes on way back up
561  bool deleteChild = deleteNodeRecurs(node->getChild(pos), depth+1, max_depth, key);
562  if (deleteChild){
563  // TODO: lazy eval?
564  node->deleteChild(pos);
565  this->tree_size-=1;
566  this->size_changed = true;
567  if (!node->hasChildren())
568  return true;
569  else{
570  node->updateOccupancyChildren();
571  }
572  }
573  // node did not lose a child, or still has other children
574  return false;
575 
576  }

712  return;
713 
714  // empty tree
715  if (root == NULL){
716  min_value[0] = min_value[1] = min_value[2] = 0.0;

725  }
