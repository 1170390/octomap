154  inline void computeChildKey (const unsigned int& pos, const unsigned short int& center_offset_key,
155  const OcTreeKey& parent_key, OcTreeKey& child_key) {
156  // x-axis
157  if (pos & 1) child_key[0] = parent_key[0] + center_offset_key;
158  else child_key[0] = parent_key[0] - center_offset_key - (center_offset_key ? 0 : 1);
159  // y-axis
160  if (pos & 2) child_key[1] = parent_key[1] + center_offset_key;
161  else child_key[1] = parent_key[1] - center_offset_key - (center_offset_key ? 0 : 1);
162  // z-axis
163  if (pos & 4) child_key[2] = parent_key[2] + center_offset_key;
164  else child_key[2] = parent_key[2] - center_offset_key - (center_offset_key ? 0 : 1);
165  }
