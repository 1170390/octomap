159  void singleIncrement(){
160  StackElement top = stack.top();
161  stack.pop();
162  if (top.depth == maxDepth)
163  return;
164 
165  StackElement s;
166  s.depth = top.depth +1;
167 
168  unsigned short int center_offset_key = tree->tree_max_val >> s.depth;
169  // push on stack in reverse order
170  for (int i=7; i>=0; --i) {
171  if (top.node->childExists(i)) {
172  computeChildKey(i, center_offset_key, top.key, s.key);
173  s.node = top.node->getChild(i);
174  //OCTOMAP_DEBUG_STR("Current depth: " << int(top.depth) << " new: "<< int(s.depth) << " child#" << i <<" ptr: "<<s.node);
175  stack.push(s);
176  assert(s.depth <= maxDepth);
177  }
178  }
