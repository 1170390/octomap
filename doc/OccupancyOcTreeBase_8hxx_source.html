00266           // not a pruned node, create requested child

00268           this->tree_size++;
00269           this->size_changed = true;
00270           created_node = true;
00271         }
00272       }

00275         return updateNodeRecurs(node->getChild(pos), created_node, key, depth+1, log_odds_update, lazy_eval);

00739   std::istream& OccupancyOcTreeBase<NODE>::readBinaryNode(std::istream &s, NODE* node) const {

00742     char child5to8_char;
00743     s.read((char*)&child1to4_char, sizeof(char));
00744     s.read((char*)&child5to8_char, sizeof(char));
00745 
00746     std::bitset<8> child1to4 ((unsigned long long) child1to4_char);
00747     std::bitset<8> child5to8 ((unsigned long long) child5to8_char);
00748 
00749     //     std::cout << "read:  "
00750     //        << child1to4.to_string<char,std::char_traits<char>,std::allocator<char> >() << " "
00751     //        << child5to8.to_string<char,std::char_traits<char>,std::allocator<char> >() << std::endl;
00752 
00753 
00754     // inner nodes default to occupied
00755     node->setLogOdds(this->clamping_thres_max);
00756 
00757     for (unsigned int i=0; i<4; i++) {
00758       if ((child1to4[i*2] == 1) && (child1to4[i*2+1] == 0)) {
00759         // child is free leaf
00760         node->createChild(i);
00761         node->getChild(i)->setLogOdds(this->clamping_thres_min);
00762       }
00763       else if ((child1to4[i*2] == 0) && (child1to4[i*2+1] == 1)) {
00764         // child is occupied leaf
00765         node->createChild(i);
00766         node->getChild(i)->setLogOdds(this->clamping_thres_max);
00767       }
00768       else if ((child1to4[i*2] == 1) && (child1to4[i*2+1] == 1)) {
00769         // child has children
00770         node->createChild(i);
00771         node->getChild(i)->setLogOdds(-200.); // child is unkown, we leave it uninitialized
00772       }
00773     }
00774     for (unsigned int i=0; i<4; i++) {
00775       if ((child5to8[i*2] == 1) && (child5to8[i*2+1] == 0)) {
00776         // child is free leaf
00777         node->createChild(i+4);
00778         node->getChild(i+4)->setLogOdds(this->clamping_thres_min);
00779       }
00780       else if ((child5to8[i*2] == 0) && (child5to8[i*2+1] == 1)) {
00781         // child is occupied leaf

00789       }

00811     std::bitset<8> child1to4;
00812     std::bitset<8> child5to8;

00815     // 01 : child is occupied node

00821     //          can be one logic expression per bit

00823     for (unsigned int i=0; i<4; i++) {

00826         if      (child->hasChildren())  { child1to4[i*2] = 1; child1to4[i*2+1] = 1; }
00827         else if (this->isNodeOccupied(child)) { child1to4[i*2] = 0; child1to4[i*2+1] = 1; }
00828         else                            { child1to4[i*2] = 1; child1to4[i*2+1] = 0; }
00829       }
00830       else {
00831         child1to4[i*2] = 0; child1to4[i*2+1] = 0;
00832       }
00833     }
00834 
00835     for (unsigned int i=0; i<4; i++) {
00836       if (node->childExists(i+4)) {
00837         const NODE* child = node->getChild(i+4);
00838         if      (child->hasChildren())  { child5to8[i*2] = 1; child5to8[i*2+1] = 1; }
00839         else if (this->isNodeOccupied(child)) { child5to8[i*2] = 0; child5to8[i*2+1] = 1; }
00840         else                            { child5to8[i*2] = 1; child5to8[i*2+1] = 0; }
00841       }
00842       else {
00843         child5to8[i*2] = 0; child5to8[i*2+1] = 0;
00844       }
00845     }
