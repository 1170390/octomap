141  template <typename T>
142  bool OcTreeDataNode<T>::collapsible() const {
143  // all children must exist, must not have children of
144  // their own and have the same occupancy probability
145  if (!childExists(0) || getChild(0)->hasChildren())
146  return false;
147 
148  T childValue = getChild(0)->getValue();
149 
150  for (unsigned int i = 1; i<8; i++) {
151  if (!childExists(i)) return false;
152  else if (getChild(i)->hasChildren()) return false;
153  else if (! (getChild(i)->getValue() == childValue)) return false;
154  }
155  return true;
156  }
