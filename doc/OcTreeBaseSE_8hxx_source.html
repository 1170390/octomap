56  template <class NODE>
57  bool OcTreeBaseSE<NODE>::computeRayKeys(const point3d& origin,
58  const point3d& end,
59  KeyRay& ray) const {
60 
61  // std::cout << "using key ray method\n";
62 
63  // see "A Faster Voxel Traversal Algorithm for Ray Tracing" by Amanatides & Woo
64  // basically: DDA in 3D
65 
66  ray.reset();
67 
68  OcTreeKey key_origin, key_end;
69  if ( !OcTreeBase<NODE>::coordToKeyChecked(origin, key_origin) ||
70  !OcTreeBase<NODE>::coordToKeyChecked(end, key_end) ) {
71  OCTOMAP_WARNING_STR("Coordinates out of bounds during ray casting");
72  return false;
73  }
74 
75  ray.addKey(key_origin);
76 
77  if (key_origin == key_end) return true; // same tree cell, we're done.
78 
79 
80  // Initialization phase -------------------------------------------------------
81 
82  point3d direction = (end - origin);
83  double length = direction.norm2();
84  direction /= length; // normalize vector
85 
86  int step[3];
87  double tMax[3];
88  double tDelta[3];
89 
90  OcTreeKey current_key = key_origin;
91 
92  for(unsigned int i=0; i < 3; ++i) {
93 
94  // compute step direction
95  if (direction(i) > 0.0) step[i] = 1;
96  else if (direction(i) < 0.0) step[i] = -1;
97  else step[i] = 0;
98 
99  // compute tMax, tDelta
100  double voxelBorder = this->keyToCoord(current_key[i]); // negative corner point of voxel
101  voxelBorder += double(step[i] * this->resolution * 0.5);
102 
103  if (step[i] != 0) {
104  tMax[i] = ( voxelBorder - origin(i) ) / direction(i);
105  tDelta[i] = this->resolution / fabs( direction(i) );
106  }
107  else {
108  tMax[i] = std::numeric_limits<double>::max();
109  tDelta[i] = std::numeric_limits<double>::max();
110  }
111  }
112 
113  // for speedup:
114  point3d origin_scaled = origin;
115  origin_scaled /= this->resolution;
116  double length_scaled = length - this->resolution/2.; // safety margin
117  length_scaled /= this->resolution; // scale
118  length_scaled = length_scaled*length_scaled; // avoid sqrt in dist comp.
119 
120  // Incremental phase ---------------------------------------------------------
121 
122  bool done = false;
123  while (!done) {
124 
125  unsigned int dim;
126 
127  // find minimum tMax:
128  if (tMax[0] < tMax[1]){
129  if (tMax[0] < tMax[2]) dim = 0;
130  else dim = 2;
131  }
132  else {
133  if (tMax[1] < tMax[2]) dim = 1;
134  else dim = 2;
135  }
136 
137  // advance in direction "dim"
138  current_key[dim] += step[dim];
139  tMax[dim] += tDelta[dim];
140 
141  assert ((current_key[dim] >= 0) && (current_key[dim] < 2*this->tree_max_val));
142 
143  // reached endpoint, key equv?
144  if (current_key == key_end) {
145  done = true;
146  break;
147  }
148  else {
149 
150  // reached endpoint world coords?
151  double dist_from_endpoint = 0;
152  for (unsigned int j = 0; j < 3; j++) {
153  double coord = (double) current_key[j] - (double) this->tree_max_val;
154  dist_from_endpoint += (coord - origin_scaled(j)) * (coord - origin_scaled(j));
155  }
156  if (dist_from_endpoint > length_scaled) {
157  done = true;
158  break;
159  }
160 
161  else { // continue to add freespace cells
162  ray.addKey(current_key);
163  }
164  }
165 
166  assert ( ray.size() < ray.sizeMax() - 1);
167 
168  } // end while
169 
170  return true;
171  }
